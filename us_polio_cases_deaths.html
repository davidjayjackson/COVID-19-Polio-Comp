US Polio Cases and Deaths by Year

polio_numbs <- read.csv("./polio-cases-and-deaths.csv")
polio_numbs <- subset(polio_numbs,Year <=1970)
summary(polio_numbs$Cases)
##    Min. 1st Qu.  Median    Mean 3rd Qu.    Max. 
##      20    1934    5342   10122   11242   57879
summary(polio_numbs$Deaths)
##    Min. 1st Qu.  Median    Mean 3rd Qu.    Max.    NA's 
##     7.0   425.8   839.0  1024.5  1378.2  7130.0       2
ggplot(polio_numbs,aes(x=Year,y=Cases)) +
  geom_col() + geom_smooth(span=0.25) +
  labs(title="US Polio Cases: 1910 - 1970")
